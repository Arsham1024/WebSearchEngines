           <a href="https://wikimedia.pl" rel="nofollow" title="Wikimedia Polska">
            <img alt="Wikimedia Polska" data-file-height="285" data-file-width="285" decoding="async" height="50" src="//upload.wikimedia.org/wikipedia/commons/thumb/8/8c/Wikimedia_Polska_logo.svg/50px-Wikimedia_Polska_logo.svg.png" srcset="//upload.wikimedia.org/wikipedia/commons/thumb/8/8c/Wikimedia_Polska_logo.svg/75px-Wikimedia_Polska_logo.svg.png 1.5x, //upload.wikimedia.org/wikipedia/commons/thumb/8/8c/Wikimedia_Polska_logo.svg/100px-Wikimedia_Polska_logo.svg.png 2x" width="50"/>

            <a class="external text" href="https://wikimedia.pl" rel="nofollow">

           </b>

          </p>

           <a href="https://meta.wikimedia.org/wiki/Strona_g%C5%82%C3%B3wna" title="Meta-wiki">
            <img alt="Meta-wiki" data-file-height="900" data-file-width="900" decoding="async" height="55" src="//upload.wikimedia.org/wikipedia/commons/thumb/7/75/Wikimedia_Community_Logo.svg/55px-Wikimedia_Community_Logo.svg.png" srcset="//upload.wikimedia.org/wikipedia/commons/thumb/7/75/Wikimedia_Community_Logo.svg/83px-Wikimedia_Community_Logo.svg.png 1.5x, //upload.wikimedia.org/wikipedia/commons/thumb/7/75/Wikimedia_Community_Logo.svg/110px-Wikimedia_Community_Logo.svg.png 2x" width="55"/>
