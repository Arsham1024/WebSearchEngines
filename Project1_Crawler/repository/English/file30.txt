         <p>
          <b>
           <a href="/wiki/%D8%A7%D8%B6%D8%B7%D8%B1%D8%A7%D8%A8_%D8%AB%D9%86%D8%A7%D8%A6%D9%8A_%D8%A7%D9%84%D9%82%D8%B7%D8%A8" title="اضطراب ثنائي القطب">
            الاضطراب ثنائي القطب
           </a>
          </b>
          هو
          <a href="/wiki/%D8%A7%D8%B6%D8%B7%D8%B1%D8%A7%D8%A8_%D9%86%D9%81%D8%B3%D9%8A" title="اضطراب نفسي">
           اضطراب نفسي
          </a>
          يسبّب نوبات من
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%83%D8%AA%D8%A6%D8%A7%D8%A8_(%D8%AD%D8%A7%D9%84%D8%A9_%D9%86%D9%81%D8%B3%D9%8A%D8%A9)" title="اكتئاب (حالة نفسية)">
           الاكتئاب
          </a>
          ونوبات من
          <a href="/wiki/%D8%A7%D8%A8%D8%AA%D9%87%D8%A7%D8%AC" title="ابتهاج">
           الابتهاج
          </a>
          غير الطبيعي. تعدّ نوبات الابتهاج المذكورة، والتي تعرف أيضاً باسم
          <a href="/wiki/%D9%87%D9%88%D8%B3" title="هوس">
           الهوس
          </a>
          أو
          <a href="/wiki/%D9%87%D9%88%D8%B3_%D8%AE%D9%81%D9%8A%D9%81" title="هوس خفيف">
           الهوس الخفيف
          </a>
          ، ذات أهمّية في تشخيص الحالة وذلك اعتماداً على شدّتها أو إذا كانت أعراض
          <a href="/wiki/%D8%B0%D9%87%D8%A7%D9%86" title="ذهان">
           الذهان
          </a>
          بادية. تختلف نوبات الابتهاج غير الطبيعي عن الشعور بالابتهاج في الظروف الاعتيادية، إذ في الحالة المرضية يشعر المرء حينها بنشاط وسعادة وتهيّج غير طبيعي؛ كما أنّها تؤدّي بالشخص في بعض الأحيان للقيام بأعمال طائشة وغير مسؤولة أو مدروسة العواقب. تتضاءل الحاجة إلى النوم أثناء نوبات الهوس؛  في حين أنّه خلال نوبات الاكتئاب قد يظهر على الأشخاص المصابين أعراض من قبيل نوبات البكاء، كذلك النظرة السوداوية للحياة، بالإضافة إلى تجنّب
          <a href="/wiki/%D8%A7%D9%84%D8%AA%D9%82%D8%A7%D8%A1_%D8%A7%D9%84%D8%B9%D9%8A%D9%88%D9%86" title="التقاء العيون">
           الالتقاء
          </a>
          مع الآخرين. يظلّ خطر
          <a href="/wiki/%D8%A7%D9%86%D8%AA%D8%AD%D8%A7%D8%B1" title="انتحار">
           الانتحار
          </a>
          عند المرضى المصابين بالاضطراب ثنائي القطب مرتفعاً، بنسبة تفوق 6%، في نفس الوقت قد تحدث أيضاً حالات من
          <a href="/wiki/%D8%A5%D9%8A%D8%B0%D8%A7%D8%A1_%D8%A7%D9%84%D9%86%D9%81%D8%B3" title="إيذاء النفس">
           إيذاء النفس
          </a>
          عند حوالي 30% إلى 40% من الحالات.  يمكن لاضطرابات نفسية أخرى مثل
          <a href="/wiki/%D8%A7%D8%B6%D8%B7%D8%B1%D8%A7%D8%A8_%D8%A7%D9%84%D9%82%D9%84%D9%82" title="اضطراب القلق">
           اضطراب القلق
          </a>
          أو
          <a href="/wiki/%D8%A7%D8%B6%D8%B7%D8%B1%D8%A7%D8%A8_%D8%AA%D8%B9%D8%A7%D8%B7%D9%8A_%D8%A7%D9%84%D9%85%D8%AE%D8%AF%D8%B1%D8%A7%D8%AA" title="اضطراب تعاطي المخدرات">
           اضطراب تعاطي المخدرات
          </a>
          أن تكون مرتبطة بالاضطراب ثنائي القطب. لا تزال مسبّبات هذا الاضطراب غير مفهومة بوضوح؛ إلّا أنّ
          <a href="/wiki/%D8%B9%D8%A7%D9%85%D9%84_%D8%A8%D9%8A%D8%A6%D9%8A" title="عامل بيئي">
           العوامل البيئية
          </a>
          <a href="/wiki/%D8%B9%D9%84%D9%85_%D8%A7%D9%84%D9%88%D8%B1%D8%A7%D8%AB%D8%A9" title="علم الوراثة">
           والوراثية
          </a>
          يمكن أن يكون لها دور في ذلك. تتضمّن العوامل البيئية الخطرة وجود
          <a href="/wiki/%D8%A5%D8%B3%D8%A7%D8%A1%D8%A9_%D9%84%D9%84%D8%A3%D8%B7%D9%81%D8%A7%D9%84" title="إساءة للأطفال">
           إساءة في مرحلة الطفولة
          </a>
          أو حالات طويلة الأمد من
          <a href="/wiki/%D8%AA%D9%88%D8%AA%D8%B1_(%D8%B9%D9%84%D9%85_%D9%86%D9%81%D8%B3)" title="توتر (علم نفس)">
           التوتّر النفسي
          </a>
          (الكرب)؛ في حين أنّ العديد من التأثيرات الجينية الصغيرة تساهم في خطر الإصابة بهذا الاضطراب النفسي؛ وتشير بعض الدراسات أنّ
          <a href="/wiki/%D9%88%D8%B1%D9%88%D8%AB%D9%8A%D8%A9" title="وروثية">
           العوامل الوراثية
          </a>
          هي الأكثر تأثيراً في الواقع، إذ يُعزى إليها حدوث حوالي 85% من الحالات. ينقسم الاضطراب ثنائي القطب بصفةٍ عامّة إلى نوعين: النوع الأول يتميّز بحدوث نوبة هوس واحدة على الأقلّ، مع أو بدون نوبات اكتئابية؛ أمّا النوع الثاني فيتميّز بوقوع نوبة واحدة على الأقل من الهوس الخفيف (وليس الهوس) ونوبة اكتئابية رئيسية (كبرى). قد تُشخّص حالات من
          <a href="/wiki/%D8%AF%D9%88%D8%B1%D9%88%D9%8A%D8%A9_%D8%A7%D9%84%D9%85%D8%B2%D8%A7%D8%AC" title="دوروية المزاج">
           اضطراب المزاج الدوروي
          </a>
          عند الأشخاص الذين يعانون من أعراض أقلّ شدّة وطويلة الأمد من الاضطراب ثنائي القطب.
         </p>

         <p>
          <b>
           <a href="/wiki/%D8%AF%D9%85%D8%A7%D8%BA_%D8%A8%D8%B4%D8%B1%D9%8A" title="دماغ بشري">
            الدماغ البشريّ
           </a>
          </b>
          هو العضو المركزي
          <a href="/wiki/%D8%AC%D9%87%D8%A7%D8%B2_%D8%B9%D8%B5%D8%A8%D9%8A" title="جهاز عصبي">
           للجهاز العصبي
          </a>
          البشري، ويُشكل مع
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AD%D8%A8%D9%84_%D8%A7%D9%84%D8%B4%D9%88%D9%83%D9%8A" title="الحبل الشوكي">
           الحبل الشوكي
          </a>
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AC%D9%87%D8%A7%D8%B2_%D8%A7%D9%84%D8%B9%D8%B5%D8%A8%D9%8A_%D8%A7%D9%84%D9%85%D8%B1%D9%83%D8%B2%D9%8A" title="الجهاز العصبي المركزي">
           الجهاز العصبي المركزي
          </a>
          . يتكون الدماغ من
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D9%85%D8%AE" title="المخ">
           المخ
          </a>
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D9%85%D8%AE%D9%8A%D8%AE" title="المخيخ">
           والمخيخ
          </a>
          <a href="/wiki/%D8%AC%D8%B0%D8%B9_%D8%A7%D9%84%D8%AF%D9%85%D8%A7%D8%BA" title="جذع الدماغ">
           والجذع الدماغي
          </a>
          . يقوم بالتحكم في معظم أنشطة الجسم، بمعالجة ودمج وتنسيق المعلومات التي يتلقاها من
          <a href="/wiki/%D8%AC%D9%87%D8%A7%D8%B2_%D8%A5%D8%AD%D8%B3%D8%A7%D8%B3" title="جهاز إحساس">
           الأجهزة الحسيِّة
          </a>
          ، ويتخذ القرارات فيما يتعلق بالتعليمات المُرسلة إلى باقي أعضاء الجسم. يتواجد الدماغ داخل
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AC%D9%85%D8%AC%D9%85%D8%A9" title="الجمجمة">
           الجمجمة
          </a>
          ومحميّ بواسطتِها. يتكون المخ وهو الجزء الأكبر مِن دماغ الإنسان. من
          <a href="/wiki/%D9%86%D8%B5%D9%81_%D9%83%D8%B1%D8%A9_%D9%85%D8%AE%D9%8A%D8%A9" title="نصف كرة مخية">
           نصفيِّ الكرة المخيِّة
          </a>
          يحتوي كل نصف كرة على لبّ داخلي يتكون من
          <a class="mw-redirect" href="/wiki/%D9%85%D8%A7%D8%AF%D8%A9_%D8%A8%D9%8A%D8%B6%D8%A7%D8%A1" title="مادة بيضاء">
           مادة بيضاء
          </a>
          . وسطح بَرّانيّ -
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D9%82%D8%B4%D8%B1%D8%A9_%D8%A7%D9%84%D9%85%D8%AE%D9%8A%D8%A9" title="القشرة المخية">
           القشرة المخية
          </a>
          - يتكون من
          <a class="mw-redirect" href="/wiki/%D9%85%D8%A7%D8%AF%D8%A9_%D8%B1%D9%85%D8%A7%D8%AF%D9%8A%D8%A9" title="مادة رمادية">
           مادة رمادية
          </a>
          . تتكون القشرة المُخية من طبقة بارنية تُسمى
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D9%82%D8%B4%D8%B1%D8%A9_%D8%A7%D9%84%D8%AD%D8%AF%D9%8A%D8%AB%D8%A9" title="القشرة الحديثة">
           القشرة الحديثة
          </a>
          وطبقة داخلية تُسمى
          <a href="/wiki/%D9%82%D8%B4%D8%B1%D8%A9_%D8%B9%D8%B1%D9%8A%D9%82%D8%A9" title="قشرة عريقة">
           القشرة العريقة
          </a>
          . تتكون القشرة المخيِّة الحديثة من سِتِّ طبقات عصبيّة، في حين تتكون القشرة العَريقة من ثلاث أو أربع طبقات فقط. ينقسم كل نصف كرة مخية إلى أربعة فصوص
          <a href="/wiki/%D9%81%D8%B5_%D8%AC%D8%A8%D9%87%D9%8A" title="فص جبهي">
           الفص جبهي
          </a>
          <a href="/wiki/%D9%81%D8%B5_%D8%B5%D8%AF%D8%BA%D9%8A" title="فص صدغي">
           والصدغي
          </a>
          <a href="/wiki/%D9%81%D8%B5_%D8%AC%D8%AF%D8%A7%D8%B1%D9%8A" title="فص جداري">
           والفص الجداري
          </a>
          <a href="/wiki/%D9%81%D8%B5_%D9%82%D8%B0%D8%A7%D9%84%D9%8A" title="فص قذالي">
           والقَذالي
          </a>
          . يرتبط الفص الجبهي بالوظائف التنفيذية بما في ذلك
          <a href="/wiki/%D8%B6%D8%A8%D8%B7_%D8%A7%D9%84%D9%86%D9%81%D8%B3" title="ضبط النفس">
           ضبط النفس
          </a>
          <a class="mw-redirect mw-disambig" href="/wiki/%D8%A7%D9%84%D8%AA%D8%AE%D8%B7%D9%8A%D8%B7" title="التخطيط">
           والتخطيط
          </a>
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AA%D9%81%D9%83%D9%8A%D8%B1" title="التفكير">
           والتفكير
          </a>
          <a href="/wiki/%D8%AA%D8%AC%D8%B1%D9%8A%D8%AF" title="تجريد">
           والتفكير المجرد
          </a>
          ، في حين أنَّ الفص القذالي مُخصص للرؤية. داخل كل فص ترتبط المناطق القشريِّةُ بوظائف محددة. على الرغم من أن نصفيّ الكرة المخيِّة الأيسر والأيمن متشابهان إلى حدٍ كبير في الشكل والوظائِف، إلا أنَّ بعض الوظائف
          <a href="/wiki/%D8%AA%D9%81%D8%A7%D8%B1%D9%82_%D9%88%D8%B8%D8%A7%D8%A6%D9%81_%D8%A7%D9%84%D8%AF%D9%85%D8%A7%D8%BA" title="تفارق وظائف الدماغ">
           مُرتبطة بجانب واحد
          </a>
          ، مثل
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D9%84%D8%BA%D8%A9" title="اللغة">
           اللغة
          </a>
          في الشق الأيسر
          <a href="/wiki/%D8%AA%D9%81%D9%83%D9%8A%D8%B1_%D9%85%D8%B1%D8%A6%D9%8A" title="تفكير مرئي">
           والقدرة البصرية والمكانية
          </a>
          في الشق الأيمن. يرتبط نصفيّ الكرة المخيِّة من خلال
          <a href="/wiki/%D8%A3%D9%84%D9%8A%D8%A7%D9%81_%D8%B5%D9%88%D8%A7%D8%B1%D9%8A%D8%A9" title="ألياف صوارية">
           السبُل العصبيّة صواريّة
          </a>
          ، أكبرها هو
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AC%D8%B3%D9%85_%D8%A7%D9%84%D8%AB%D9%81%D9%86%D9%8A" title="الجسم الثفني">
           الجسم الثفني
          </a>
          . يرتبط المُخ بالحبل الشوكيّ بواسطة الجِذع الدماغي. يتكون جذع الدماغ من
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AF%D9%85%D8%A7%D8%BA_%D8%A7%D9%84%D9%85%D8%AA%D9%88%D8%B3%D8%B7" title="الدماغ المتوسط">
           الدماغ المتوسط
          </a>
          <a href="/wiki/%D8%AC%D8%B3%D8%B1_(%D8%AA%D8%B4%D8%B1%D9%8A%D8%AD)" title="جسر (تشريح)">
           والجسر
          </a>
          <a href="/wiki/%D9%86%D8%AE%D8%A7%D8%B9_%D9%85%D8%B3%D8%AA%D8%B7%D9%8A%D9%84" title="نخاع مستطيل">
           والنخاع المستطيل
          </a>
          . يرتبط المُخيخ بجِذْع الدِّماغ عن طريق
          <a href="/wiki/%D8%B3%D9%88%D9%8A%D9%82%D8%A9_%D9%85%D8%AE%D9%8A%D8%AE%D9%8A%D8%A9" title="سويقة مخيخية">
           السويقات المُخيخيِّة
          </a>
          . يوجد داخل المخ
          <a href="/wiki/%D8%AC%D9%87%D8%A7%D8%B2_%D8%A8%D8%B7%D9%8A%D9%86%D9%8A" title="جهاز بطيني">
           الجهاز البطينيّ
          </a>
          الذي يتكون من أربعة بطينات مُترابطة تقوم بإنتاج
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%B3%D8%A7%D8%A6%D9%84_%D8%A7%D9%84%D8%AF%D9%85%D8%A7%D8%BA%D9%8A_%D8%A7%D9%84%D8%B4%D9%88%D9%83%D9%8A" title="السائل الدماغي الشوكي">
           السائل الدماغي الشوكي
          </a>
          وتوزيعِه. يتواجد تحت القشرة المُخية العديد من البِنيات الهامة، بما في ذلك
          <a href="/wiki/%D9%85%D9%87%D8%A7%D8%AF" title="مهاد">
           المِهاد
          </a>
          <a href="/wiki/%D9%81%D9%88%D9%82_%D8%A7%D9%84%D9%85%D9%87%D8%A7%D8%AF" title="فوق المهاد">
           وفوق المهاد
          </a>
          <a href="/wiki/%D8%BA%D8%AF%D8%A9_%D8%B5%D9%86%D9%88%D8%A8%D8%B1%D9%8A%D8%A9" title="غدة صنوبرية">
           والغُدةُ الصَّنوبريَّة
          </a>
          <a href="/wiki/%D8%AA%D8%AD%D8%AA_%D8%A7%D9%84%D9%85%D9%87%D8%A7%D8%AF" title="تحت المهاد">
           وتحت المهاد
          </a>
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%BA%D8%AF%D8%A9_%D8%A7%D9%84%D9%86%D8%AE%D8%A7%D9%85%D9%8A%D8%A9" title="الغدة النخامية">
           والغدة النخامية
          </a>
          <a href="/wiki/%D9%85%D9%87%D8%A7%D8%AF_%D8%B3%D9%81%D9%84%D9%8A" title="مهاد سفلي">
           والمهاد السفليّ
          </a>
          <a href="/wiki/%D8%AC%D9%87%D8%A7%D8%B2_%D9%86%D8%B7%D8%A7%D9%82%D9%8A" title="جهاز نطاقي">
           والجهاز الحُوفِيّ
          </a>
          ، بما في ذلك
          <a href="/wiki/%D8%A7%D9%84%D9%84%D9%88%D8%B2%D8%A9" title="اللوزة">
           اللوزة
          </a>
          <a href="/wiki/%D9%82%D8%B1%D9%86_%D8%A2%D9%85%D9%88%D9%86" title="قرن آمون">
           وقرن آمون
          </a>
          <a href="/wiki/%D8%B9%D8%A7%D8%A6%D9%82_(%D8%B7%D8%A8)" title="عائق (طب)">
           والعائق
          </a>
          <a href="/wiki/%D9%86%D9%88%D8%A7%D8%A9_(%D8%AA%D8%B4%D8%B1%D9%8A%D8%AD_%D8%B9%D8%B5%D8%A8%D9%8A)" title="نواة (تشريح عصبي)">
           والنواة
          </a>
          المختلفة
          <a href="/wiki/%D8%B9%D9%82%D8%AF_%D9%82%D8%A7%D8%B9%D8%AF%D9%8A%D8%A9" title="عقد قاعدية">
           للعُقد القاعديِّة
          </a>
          ؛ بنيات
          <a href="/wiki/%D8%AF%D9%85%D8%A7%D8%BA_%D8%A3%D9%85%D8%A7%D9%85%D9%8A_%D9%82%D8%A7%D8%B9%D8%AF%D9%8A" title="دماغ أمامي قاعدي">
           الدماغ الأمامي القاعدي
          </a>
          وثلاثة
          <a href="/wiki/%D8%A3%D8%B9%D8%B6%D8%A7%D8%A1_%D9%85%D8%AD%D9%8A%D8%B7%D8%A9_%D8%A8%D8%A7%D9%84%D8%A8%D8%B7%D9%8A%D9%86%D8%A7%D8%AA" title="أعضاء محيطة بالبطينات">
           أعضاء محيطة بالبطينات
          </a>
          . تشمل خلايا الدماغ
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AE%D9%84%D8%A7%D9%8A%D8%A7_%D8%A7%D9%84%D8%B9%D8%B5%D8%A8%D9%8A%D8%A9" title="الخلايا العصبية">
           الخلايا العصبية
          </a>
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D8%AE%D9%84%D8%A7%D9%8A%D8%A7_%D8%A7%D9%84%D8%AF%D8%A8%D9%82%D9%8A%D8%A9" title="الخلايا الدبقية">
           والخلايا الدبقية
          </a>
          المُساعدة. هناك أكثر من 86 مليار خليّة عصبيِّة في الدماغ، وعدد متساوٍ إلى حدٍ ما من الخلايا الأخرى. يكون نشاط الدماغ ممكنًا من خلال ترابط الخلايا العصبيِّة وإطلاقها
          <a class="mw-redirect" href="/wiki/%D8%A7%D9%84%D9%86%D9%88%D8%A7%D9%82%D9%84_%D8%A7%D9%84%D8%B9%D8%B5%D8%A8%D9%8A%D8%A9" title="النواقل العصبية">
           للناقلات العصبية
          </a>
          استجابةً
          <a href="/wiki/%D8%AC%D9%87%D8%AF_%D8%A7%D9%84%D9%81%D8%B9%D9%84" title="جهد الفعل">

          </a>
          . تَتصِل الخلايا العصبيّة لتشكيل
          <a href="/wiki/%D9%85%D8%B3%D8%A7%D8%B1_%D8%B9%D8%B5%D8%A8%D9%8A" title="مسار عصبي">

         <p>
          <b>
           <a href="/wiki/%D8%AC%D8%A7%D8%A6%D8%B2%D8%A9_%D8%B1%D9%88%D8%AB_%D9%84%D9%8A%D8%AA%D9%84_%D8%B3%D8%A7%D8%AA%D8%B1_%D9%81%D9%8A_%D8%A7%D9%84%D8%B1%D9%8A%D8%A7%D8%B6%D9%8A%D8%A7%D8%AA" title="جائزة روث ليتل ساتر في الرياضيات">
            جائزة روث ليتل ساتر في الرياضيات
           </a>
          </b>
          وتُسمى اختصارًا
          <b>
           جائزة ساتر
          </b>
          ، هي جائزةٌ من ضمن 21 جائزةٍ أُخرى تمنحها
          <a href="/wiki/%D8%AC%D9%85%D8%B9%D9%8A%D8%A9_%D8%A7%D9%84%D8%B1%D9%8A%D8%A7%D8%B6%D9%8A%D8%A7%D8%AA_%D8%A7%D9%84%D8%A3%D9%85%D8%B1%D9%8A%D9%83%D9%8A%D8%A9" title="جمعية الرياضيات الأمريكية">
           جمعية الرياضيات الأمريكية
          </a>
          (
          <abbr class="abbr" title="American Mathematical Society">
           AMS
          </abbr>
          ). تُمنح الجائزة كل عامين تقديرًا لمساهمةٍ بارزةٍ في أبحاث الرياضيات قامت بها امرأةٌ في السنوات الست السابقة. تأسست الجائزة في عام 1990 عبر تبرعٍ من
          <a href="/wiki/%D8%AC%D9%88%D8%A7%D9%86_%D8%A8%D9%8A%D8%B1%D9%85%D8%A7%D9%86" title="جوان بيرمان">
           جوان بيرمان
          </a>
          ، وذلك تخليدًا لذكرى أختها
          <a href="/wiki/%D8%B1%D9%88%D8%AB_%D9%84%D9%8A%D8%AA%D9%84_%D8%B3%D8%A7%D8%AA%D8%B1" title="روث ليتل ساتر">
           روث ليتل ساتر
          </a>
          ، والتي عملت أساسًا في العلوم الحيوية، وكانت مؤيدةً لضرورة تحقيق تكافؤ الفرص للمرأة في مجال العلوم. مُنحت الجائزة للمرة الأولى عام 1991، وتهدفُ إلى تقدير التزامِ روث ساتر بالأبحاث وتشجيع النساء في مجال العلوم. يتولى مجلس جمعية الرياضيات الأمريكية اختيار الفائز، وذلك اعتمادًا على توصية لجنة الاختيار. تُمنح الجائزة في
          <a href="/wiki/%D8%A7%D8%AC%D8%AA%D9%85%D8%A7%D8%B9%D8%A7%D8%AA_%D8%A7%D9%84%D8%B1%D9%8A%D8%A7%D8%B6%D9%8A%D8%A7%D8%AA_%D8%A7%D9%84%D9%85%D8%B4%D8%AA%D8%B1%D9%83%D8%A9" title="اجتماعات الرياضيات المشتركة">
           اجتماعات الرياضيات المشتركة
          </a>
          خلال السنوات
          <a class="mw-redirect" href="/wiki/%D8%A3%D8%B9%D8%AF%D8%A7%D8%AF_%D8%B2%D9%88%D8%AC%D9%8A%D8%A9_%D9%88%D8%A3%D8%B9%D8%AF%D8%A7%D8%AF_%D9%81%D8%B1%D8%AF%D9%8A%D8%A9" title="أعداد زوجية وأعداد فردية">
           الفردية
          </a>
          ، وتتضمن أيضًا منح مكافأةٍ نقديةٍ مُتواضعة. منذ عام 2003 أصبحت الجائزة تُقدر بـ5,000
          <a href="/wiki/%D8%AF%D9%88%D9%84%D8%A7%D8%B1_%D8%A3%D9%85%D8%B1%D9%8A%D9%83%D9%8A" title="دولار أمريكي">
           دولار أمريكي
          </a>
          ، بينما في الفترة بين 1997 إلى 2001، كانت تُقدر بـ1,200 دولار أمريكي، وقبل ذلك كان قدرها 4,000 دولار أمريكي. إذا مُنحت الجائزة لأكثر من شخصٍ، فإنَّ أموال الجائزة تُقسم بين الحاصلين عليها. تعتبر
          <a href="/wiki/%D8%AF%D9%88%D8%B3%D8%A7_%D9%85%D8%A7%D9%83%D8%AF%D9%88%D9%81" title="دوسا ماكدوف">
           دوسا ماكدوف
          </a>
          أول حاصلٍ على الجائزة، وذلك عن عملها في
          <a href="/wiki/%D9%87%D9%86%D8%AF%D8%B3%D8%A9_%D8%AA%D9%85%D8%A7%D8%B3%D9%83%D9%8A%D8%A9" title="هندسة تماسكية">
           الهندسة التماسكية
          </a>
          .

            <span class="mp-h2">
             ما هي ويكيبيديا؟
            </span>
           </p>
          </div>
         </div>
         <p>
          <b>
           <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%B9%D9%86" title="ويكيبيديا:عن">
            ويكيبيديا
           </a>
          </b>
          مشروع تعاوني متعدد اللغات يضم
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A" title="ويكي">
           ويكيات
          </a>
          بأكثر من 300 لغة للعمل في مشاريع
          <a href="/wiki/%D9%85%D9%88%D8%B3%D9%88%D8%B9%D8%A9" title="موسوعة">
           موسوعات
          </a>
          حرة ودقيقة ومتكاملة ومتنوعة ومحايدة، يستطيع الجميع المساهمة في تحريرها. نشأت ويكيبيديا في عام 2001، حيث نمت وتطورت بسرعة لتصبح واحدة من أكبر المواقع على الإنترنت. بدأت
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7_%D8%A7%D9%84%D8%B9%D8%B1%D8%A8%D9%8A%D8%A9" title="ويكيبيديا العربية">
           النسخة العربية
          </a>
          في
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%AA%D8%A7%D8%B1%D9%8A%D8%AE_%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7_%D8%A7%D9%84%D8%B9%D8%B1%D8%A8%D9%8A%D8%A9" title="ويكيبيديا:تاريخ ويكيبيديا العربية">
           يوليو/تموز 2003
          </a>
          .
          <br/>
          لدى ويكيبيديا بعض
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%B3%D9%8A%D8%A7%D8%B3%D8%A7%D8%AA_%D9%88%D8%A5%D8%B1%D8%B4%D8%A7%D8%AF%D8%A7%D8%AA" title="ويكيبيديا:سياسات وإرشادات">
           السياسات والإرشادات
          </a>
          التي تساعد متطوعيها على العمل فيها، بعض هذه السياسات لا تزال في طور التشكل، بينما استقرت سياسات أخرى منذ زمن، ومع أن سياسات ويكيبيديا مستمرة في النمو، إلا أن بعض
          <a href="/wiki/%D9%85%D8%AC%D8%AA%D9%85%D8%B9_%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7" title="مجتمع ويكيبيديا">
           الويكيبيديين
          </a>
          يشعرون أن كتابة القواعد أمر لا يمكن له أن يغطي كل تنويعة على السلوك المثير أو الهادف لضرر. هؤلاء الذين يحررون
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A7%D9%81%D8%AA%D8%B1%D8%B6_%D8%AD%D8%B3%D9%86_%D8%A7%D9%84%D9%86%D9%8A%D8%A9" title="ويكيبيديا:افترض حسن النية">
           بنية حسنة
          </a>
          ويظهرون
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A3%D8%AF%D8%A8" title="ويكيبيديا:أدب">
           سلوكًا مدنيًّا
          </a>
          ويسعون وراء
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%AA%D9%88%D8%A7%D9%81%D9%82" title="ويكيبيديا:توافق">
           التوافق
          </a>
          ويعملون لتحقيق هدفهم بالمشاركة في مشروع
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D9%88%D8%AC%D9%87%D8%A9_%D8%A7%D9%84%D9%86%D8%B8%D8%B1_%D8%A7%D9%84%D9%85%D8%AD%D8%A7%D9%8A%D8%AF%D8%A9" title="ويكيبيديا:وجهة النظر المحايدة">
           موسوعي محايد
          </a>
          ، سيجدون أنفسهم في
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A3%D8%AF%D8%A8" title="ويكيبيديا:أدب">
           بيئة مرحبة
          </a>
          .
          <br/>
          يمكنك دائمًا المساعدة في بناء ويكيبيديا وتحسينها بالتعاون مع
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A8%D9%88%D8%A7%D8%A8%D8%A9_%D8%A7%D9%84%D9%85%D8%AC%D8%AA%D9%85%D8%B9" title="ويكيبيديا:بوابة المجتمع">
           المجتمع
          </a>
          عن طريق
          <a href="/wiki/%D9%85%D8%B3%D8%A7%D8%B9%D8%AF%D8%A9:%D8%AA%D8%AD%D8%B1%D9%8A%D8%B1_%D8%A7%D9%84%D8%B5%D9%81%D8%AD%D8%A7%D8%AA" title="مساعدة:تحرير الصفحات">
           التحرير
          </a>
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D9%85%D9%82%D8%A7%D9%84%D8%AA%D9%83_%D8%A7%D9%84%D8%A3%D9%88%D9%84%D9%89" title="ويكيبيديا:مقالتك الأولى">
           وإنشاء مقالات جديدة
          </a>
          . انظر
          <a href="/wiki/%D9%85%D8%B3%D8%A7%D8%B9%D8%AF%D8%A9:%D9%85%D8%AD%D8%AA%D9%88%D9%8A%D8%A7%D8%AA" title="مساعدة:محتويات">
           قسم المساعدة
          </a>
          للحصول على نصائح، أو توجه إلى
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A7%D9%84%D9%85%D9%8A%D8%AF%D8%A7%D9%86" title="ويكيبيديا:الميدان">
           الميدان
          </a>
          لطرح الاستفسارات العامة حول أساليب التحرير وسياسات ويكيبيديا والقضايا التقنية أو اللغوية.
         </p>

           <p>
            <span class="mp-h2">
             ما هي ويكيبيديا؟
            </span>
           </p>
          </div>
         </div>
         <p>
          <b>
           <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%B9%D9%86" title="ويكيبيديا:عن">
            ويكيبيديا
           </a>
          </b>
          مشروع تعاوني متعدد اللغات يضم
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A" title="ويكي">
           ويكيات
          </a>
          بأكثر من 300 لغة للعمل في مشاريع
          <a href="/wiki/%D9%85%D9%88%D8%B3%D9%88%D8%B9%D8%A9" title="موسوعة">
           موسوعات
          </a>
          حرة ودقيقة ومتكاملة ومتنوعة ومحايدة، يستطيع الجميع المساهمة في تحريرها. نشأت ويكيبيديا في عام 2001، حيث نمت وتطورت بسرعة لتصبح واحدة من أكبر المواقع على الإنترنت. بدأت
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7_%D8%A7%D9%84%D8%B9%D8%B1%D8%A8%D9%8A%D8%A9" title="ويكيبيديا العربية">
           النسخة العربية
          </a>
          في
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%AA%D8%A7%D8%B1%D9%8A%D8%AE_%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7_%D8%A7%D9%84%D8%B9%D8%B1%D8%A8%D9%8A%D8%A9" title="ويكيبيديا:تاريخ ويكيبيديا العربية">
           يوليو/تموز 2003
          </a>
          .
          <br/>
          لدى ويكيبيديا بعض
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%B3%D9%8A%D8%A7%D8%B3%D8%A7%D8%AA_%D9%88%D8%A5%D8%B1%D8%B4%D8%A7%D8%AF%D8%A7%D8%AA" title="ويكيبيديا:سياسات وإرشادات">
           السياسات والإرشادات
          </a>
          التي تساعد متطوعيها على العمل فيها، بعض هذه السياسات لا تزال في طور التشكل، بينما استقرت سياسات أخرى منذ زمن، ومع أن سياسات ويكيبيديا مستمرة في النمو، إلا أن بعض
          <a href="/wiki/%D9%85%D8%AC%D8%AA%D9%85%D8%B9_%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7" title="مجتمع ويكيبيديا">
           الويكيبيديين
          </a>
          يشعرون أن كتابة القواعد أمر لا يمكن له أن يغطي كل تنويعة على السلوك المثير أو الهادف لضرر. هؤلاء الذين يحررون
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A7%D9%81%D8%AA%D8%B1%D8%B6_%D8%AD%D8%B3%D9%86_%D8%A7%D9%84%D9%86%D9%8A%D8%A9" title="ويكيبيديا:افترض حسن النية">
           بنية حسنة
          </a>
          ويظهرون
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A3%D8%AF%D8%A8" title="ويكيبيديا:أدب">
           سلوكًا مدنيًّا
          </a>
          ويسعون وراء
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%AA%D9%88%D8%A7%D9%81%D9%82" title="ويكيبيديا:توافق">
           التوافق
          </a>
          ويعملون لتحقيق هدفهم بالمشاركة في مشروع
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D9%88%D8%AC%D9%87%D8%A9_%D8%A7%D9%84%D9%86%D8%B8%D8%B1_%D8%A7%D9%84%D9%85%D8%AD%D8%A7%D9%8A%D8%AF%D8%A9" title="ويكيبيديا:وجهة النظر المحايدة">
           موسوعي محايد
          </a>
          ، سيجدون أنفسهم في
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A3%D8%AF%D8%A8" title="ويكيبيديا:أدب">
           بيئة مرحبة
          </a>
          .
          <br/>
          يمكنك دائمًا المساعدة في بناء ويكيبيديا وتحسينها بالتعاون مع
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A8%D9%88%D8%A7%D8%A8%D8%A9_%D8%A7%D9%84%D9%85%D8%AC%D8%AA%D9%85%D8%B9" title="ويكيبيديا:بوابة المجتمع">
           المجتمع
          </a>
          عن طريق
          <a href="/wiki/%D9%85%D8%B3%D8%A7%D8%B9%D8%AF%D8%A9:%D8%AA%D8%AD%D8%B1%D9%8A%D8%B1_%D8%A7%D9%84%D8%B5%D9%81%D8%AD%D8%A7%D8%AA" title="مساعدة:تحرير الصفحات">
           التحرير
          </a>
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D9%85%D9%82%D8%A7%D9%84%D8%AA%D9%83_%D8%A7%D9%84%D8%A3%D9%88%D9%84%D9%89" title="ويكيبيديا:مقالتك الأولى">
           وإنشاء مقالات جديدة
          </a>
          . انظر
          <a href="/wiki/%D9%85%D8%B3%D8%A7%D8%B9%D8%AF%D8%A9:%D9%85%D8%AD%D8%AA%D9%88%D9%8A%D8%A7%D8%AA" title="مساعدة:محتويات">
           قسم المساعدة
          </a>
          للحصول على نصائح، أو توجه إلى
          <a href="/wiki/%D9%88%D9%8A%D9%83%D9%8A%D8%A8%D9%8A%D8%AF%D9%8A%D8%A7:%D8%A7%D9%84%D9%85%D9%8A%D8%AF%D8%A7%D9%86" title="ويكيبيديا:الميدان">
           الميدان
          </a>
          لطرح الاستفسارات العامة حول أساليب التحرير وسياسات ويكيبيديا والقضايا التقنية أو اللغوية.
         </p>
