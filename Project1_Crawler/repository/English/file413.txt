             <a href="/wiki/%22%D0%9E%D1%81%D0%BA%D0%B0%D1%80%22_%D1%81%D1%8B%D0%B9%D0%BB%D1%8B%D0%B3%D1%8B" title='"Оскар" сыйлыгы'>
