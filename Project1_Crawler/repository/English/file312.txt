         <a class="mw-redirect" href="/wiki/%D9%86%D8%A7%D9%88%DB%95%DA%95%DB%86%DA%A9%DB%8C_%D8%A6%D8%A7%D8%B2%D8%A7%D8%AF" title="ناوەڕۆکی ئازاد">

         <a href="/wiki/%DA%A9%D9%88%D8%B1%D8%AF%DB%8C%DB%8C_%D9%86%D8%A7%D9%88%DB%95%D9%86%D8%AF%DB%8C" title="کوردیی ناوەندی">

         <a href="/wiki/%D9%88%DB%8C%DA%A9%DB%8C%D9%BE%DB%8C%D8%AF%DB%8C%D8%A7:%D9%BE%DB%8E%D9%86%D8%AC_%DA%A9%DB%86%DA%B5%DB%95%DA%A9%DB%95" title="ویکیپیدیا:پێنج کۆڵەکە">
          سیاسەتەکان
         </a>
         پەیڕەوی دەکرێ نەک لە ھیچ تاکە کەسێک. بۆ باشترکردنی وتارەکان بەکارھێنەران ھەردەم قسەوباس و خاڵەکانی خۆیان لە
         <a href="/wiki/%DB%8C%D8%A7%D8%B1%D9%85%DB%95%D8%AA%DB%8C:%D8%A8%DB%95%DA%A9%D8%A7%D8%B1%DA%BE%DB%8E%D9%86%D8%A7%D9%86%DB%8C_%D9%BE%DB%95%DA%95%DB%95%DA%A9%D8%A7%D9%86%DB%8C_%D9%88%D8%AA%D9%88%D9%88%DB%8E%DA%98" title="یارمەتی:بەکارھێنانی پەڕەکانی وتووێژ">
          پەڕەکانی وتووێژی تایبەتدا
         </a>
         دەردەبڕن. ویکیپیدیای کوردی پێشوازی لە بیروڕاکان و وتووێژەکانی ھەموو بەکارھێنەرێک دەکات سەبارەت بە ھەموو بەشەکانی.

         ھەنووکە لە ویکیپیدیای
         <a href="/wiki/%DA%A9%D9%88%D8%B1%D8%AF%DB%8C%DB%8C_%D9%86%D8%A7%D9%88%DB%95%D9%86%D8%AF%DB%8C" title="کوردیی ناوەندی">

         <b>
          <a href="/wiki/%D8%AA%D8%A7%DB%8C%D8%A8%DB%95%D8%AA:%D8%A6%D8%A7%D9%85%D8%A7%D8%B1%DB%95%DA%A9%D8%A7%D9%86" title="تایبەت:ئامارەکان">
           ٣٤٬٨٠٢
          </a>
         </b>
         وتار ھەن، ئەمەش ئەوە دەگەیەنێت کە ئەو وتارەی کە تۆ دەتەوێت لە ویکیپیدیا بی نووسیت ئەگەری ھەبوونی ھەیە.
         <a href="/wiki/%D8%AA%D8%A7%DB%8C%D8%A8%DB%95%D8%AA:%DA%AF%DB%95%DA%95%D8%A7%D9%86" title="تایبەت:گەڕان">
          پەڕەی گەڕان
         </a>
         بەکاربھێنە بۆ ئەوەی بزانیت کە ئەو وتارەی دەتەوێت بی نووسیت ھەیە لە ویکیپیدیا یان نا، لەوانەیە لەژێر ناوێکی تر ھەبێت. ئەگەر وتارەکە ھەبوو ئەوا پێویست ناکات پەڕەیەکی نوێ دروست بکەیتەوە لەبارەی ھەمان بابەتەوە،
         <a href="/wiki/%D9%88%DB%8C%DA%A9%DB%8C%D9%BE%DB%8C%D8%AF%DB%8C%D8%A7:%D8%B3%DB%8C%D8%A7%D8%B3%DB%95%D8%AA%DB%8C_%D8%B3%DA%95%DB%8C%D9%86%DB%95%D9%88%DB%95" title="ویکیپیدیا:سیاسەتی سڕینەوە">
          چوونکە دەسڕدرێتەوە
         </a>
         . بەڵام تۆ ئازادیت لە چاکسازیکردن لەو وتارە ھەرچۆنێک بێت. ئەگەریش نەبوو، ناوی وتاری خوازراو لەم سندووقەی خوارەوە بنووسە و دروستی بکە. بیرت نەچێت کە بە ئاگایییەوە دوای
         <a href="/wiki/%D9%88%DB%8C%DA%A9%DB%8C%D9%BE%DB%8C%D8%AF%DB%8C%D8%A7:%D8%B3%DB%8C%D8%A7%D8%B3%DB%95%D8%AA%DB%95%DA%A9%D8%A7%D9%86_%D9%88_%DA%95%DB%8E%D8%B3%D8%A7%DA%A9%D8%A7%D9%86" title="ویکیپیدیا:سیاسەتەکان و ڕێساکان">
          ڕێنمایییەکان بکەویت
         </a>
         و سیاسەتەکان لەسەر
         <a href="/wiki/%D9%88%DB%8C%DA%A9%DB%8C%D9%BE%DB%8C%D8%AF%DB%8C%D8%A7:%D9%88%D8%AA%D8%A7%D8%B1_%DA%86%DB%8C%DB%8C%DB%95%D8%9F" title="ویکیپیدیا:وتار چییە؟">
          نووسینی وتار و شێوازەکانی بخوێنیتەوە.
